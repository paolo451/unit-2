```.py
## This program turns an input like "010!011!000" into its corresponding 
## decimal equivalent, where the exclamation indicates to start over the conversion

def reverseMM(numbs):
    answ = []
    prog = numbs.split("!")

    for i in range(3):
        binnum = prog[i]
        exponent = 1
        res = 0

        for digit in binnum[::-1]:
            res += int(digit) * exponent
            exponent *= 2

        answ.append(str(res))
        
    return "".join(answ)
```
